25 /**
26  * @brief Provides structure for reading and writing to shared memory segments
27  * using PHPs shmop_* functions.
28  *
29  * Two shared memory segments are opened or created if they don't already exist.
30  * The first segment is for storing an index of offsets, which references the second
31  * segment where the actual data is stored.
32  *
33  * The index segment contains a 12 byte header containing a version number,
34  * next index offset (the offset at which a new index can be written to the index
35  * segment) and next data offset (the offset at which a new data item can be
36  * written to the data segment). These three values are unsigned 32 bit integers.
37  *
38  * Derived classes should implement the following functionality:
39  *
40  * - An index structure which creates a unique identifier for a data item to be
41  * stored, that item's offset (in bytes) in the data segment, the type of data
42  * being stored and its length. For example, an index may has a structrue:
43  * name (char)
44  * type (char)
45  * length (uint16)
46  * offset (uint32)
47  *
48  * Resulting in an index that looks like:
49  * 'a C 1 0' (the type C referrs to an unsigned char, see: http://php.net/manual/en/function.pack.php)
50  * which says that data item 'a' is a char stored at offset 0 in the data segment.
51  * The value for item 'a' can then be read and written using shmop_read and shmop_write
52  * using the length and offset values.
53  * - A method for searching through the index to find the offset and length for
54  * a value based on its unique identifer. Using the above index as an example,
55  * looking up 'a' would return an offset of 0 and a length of 1, and optionally
56  * store the offset value somewhere to save further lookups.
57  * - A method for adding a value to the index.
58  * - Implementation of the __get and __set functions which use shmop_read and
59  * shmop_write on the data segment ($shmDataId), using the offset and length
60  * values of the value being read or written. A possible implementation of the
61  * __get function:
62  * lookup the offset and length of the value
63  * return data of the values length at the values offset in the data segment
64  * or false if the value doesn't exist in the index
65  * A possible implementation of the __set function:
66  * lookup the offset and length of the value
67  * if the value doesn't exist add it to the index
68  * write data of the values length at the values offset in the data segment
69  *
70  * Shared memory segments require a System V ID, which is created using ftok(),
71  * using a temporary file stored in IPC_KEY_FILE_PATH named, using the $name and
72  * $identifier passed to the constructor.
73  *
74  * This class extends stdClass. By overriding the __get and __set functions, any
75  * operations on properties that aren't explicilty defined as properties of this
76  * class (or any derived classes), will result in invokation of the __get and
77  * possibly __set functions being called (depending on the operation). This allows
78  * data to be read and written from shared memory using the following notation:
79  * @code
80  * $object->foo; // (invokes __get('foo'))
81  * $object->foo = 'baa'; // (invokes __set('foo', 'baa'))
82  * $object->counter++; // (invokes __get('counter') followed by __set('foo', n))
83  * // where n is the return of the __get + 1.
84  * @endcode
85  *
86  * This class uses file locking (flock) on the IPC key file to ensure data
87  * consistency. File locking is performed in a non blocking fashion to prevent
88  * deadlocks, with a lock wait timeout of 100 milliseconds. If a lock cannot be
89  * obtained within the timeout the operation is aborted.
90  *
91  * @todo Change the IPC_KEY_FILE_PATH constant to a variable and allow for
92  * derived classes to set this value.
93  * @todo Change the LOCK_WAIT_TIME constant to a variable and allow for
94  * derived classes to set this value.
95  *
96  * @see metrics/MetricsLogger.class.php, an implementation of SharedMemoryOp.
97  * @see Packing.class.php, for helper functions and constants when working
98  * with PHP's pack and unpack functions.
99  * @see http://php.net/manual/en/function.ftok.php
100  * @see http://www.php.net/manual/en/book.shmop.php
101  * @see http://www.php.net/manual/en/function.pack.php
102  * @see http://www.php.net/manual/en/function.unpack.php
103  */

185  /**
186  * @brief Initializes a shared memory segment.
187  *
188  * Creates an System V IPC Key file in /var/tmp if it doesn't already exist,
189  * attempts to obtain an exclusive lock on that file, opens the index shared
190  * memory segment and the data shared memory segment, sets or upgrades the
191  * version and sets the index offset and the data offset if not already set.
192  *
193  * @return True on successful initialization, false otherwise.
194  */
195  protected function initializeSharedMemorySegment()
196  {
197  if (!extension_loaded('shmop')) {
198  $this->hasError = true;
199  $this->logger->error('shmop extension not loaded');
200  return false;
201  }
202 
203  if (!file_exists($this->ipcKeyFile)) {
204  if (!$this->readOnly) {
205  try {
206  touch($this->ipcKeyFile);
207  } catch (Exception $e) {
208  $this->hasError = true;
209  $this->logger->error('Failed to touch ' . $this->ipcKeyFile);
210  return false;
211  }
212  } else {
213  $this->hasError = true;
214  $this->logger->error('File does not exist ' . $this->ipcKeyFile);
215  return false;
216  }
217  }
218 
219  $ipcKeyFileHandle = fopen($this->ipcKeyFile, 'r');
220  if ($ipcKeyFileHandle !== false) {
221  try {
222  $this->shmIndexId = $this->openOrCreate(self::FTOK_PROJECT_INDEX_IDENTIFIER, $this->shmIndexPages);
223  $this->shmDataId = $this->openOrCreate(self::FTOK_PROJECT_DATA_IDENTIFIER, $this->shmDataPages);
224  if ($this->shmIndexId !== false && $this->shmDataId !== false) {
225  $head = $this->getHead();
226  if (!$this->readOnly && ($head['version'] == 0 || $head['version'] < $this->version)) {
227  if ($this->getLock($ipcKeyFileHandle, LOCK_EX)) {
228  $head = $this->getHead();
229  if ($head['version'] == 0) {
230  // Set the version if not already set.
231  $data = pack(
232  Packing::getPackFormat('head', $this->getHeadStructure()),
233  $this->version,
234  Packing::getPackLength($this->getHeadStructure()),
235  0
236  );
237  } elseif ($head['version'] < $this->version) {
238  // Upgrade the version if an older version exists.
239  $data = pack($this->versionStructure['version'], $this->version);
240  }
241  shmop_write($this->shmIndexId, $data, 0);
242  flock($ipcKeyFileHandle, LOCK_UN);
243  fclose($ipcKeyFileHandle);
244  return true;
245  } else {
246  $this->logger->error('Failed to obtain a lock on ' . $this->ipcKeyFile);
247  }
248  } else {
249  return true;
250  }
251  }
252  } catch (Exception $e) {
253  // Let any exceptions fall through, and close the file
254  $this->logger->error($e->getMessage());
255  }
256  } else {
257  $this->logger->error('Failed to open ' . $this->ipcKeyFile);
258  }
259  if ($this->shmIndexId !== false) {
260  shmop_close($this->shmIndexId);
261  }
262  if ($this->shmDataId !== false) {
263  shmop_close($this->shmDataId);
264  }
265  flock($ipcKeyFileHandle, LOCK_UN);
266  fclose($ipcKeyFileHandle);
267  return false;
268  }

270  /**
271  * @brief Opens or creates a shared memory segment.
272  *
273  * Attempts to open a shared memory segment for the specifed project identifier.
274  * If the segment cannot be opened, attempts to create the shared memory segment.
275  *
276  * @param $projectIdentifier Project identifier to use in ftok call.
277  * @param $pages Size of shared memory segment if being created.
278  *
279  * @return An shared memory segment ID on success, false otherwise.
280  */
281  protected function openOrCreate($projectIdentifier, $pages)
282  {
283  $key = ftok($this->ipcKeyFile, $projectIdentifier);
284  if (is_long($key) && $key != -1) {
285  $mode = ($this->readOnly) ? 'a' : 'w';
286  @$shmId = shmop_open($key, $mode, 0, 0);
287  if ($shmId !== false) {
288  $this->logger->debug('Opened shared memory segment 0x' . dechex($key));
289  return $shmId;
290  } else {
291  if (!$this->readOnly) {
292  $shmId = shmop_open($key, 'c', self::SHARED_MEMORY_MODE, $pages * self::PAGE_SIZE);
293  if ($shmId !== false) {
294  $this->logger->info('Created and opened shared memory segment 0x' . dechex($key));
295  return $shmId;
296  }
297  }
298  }
299  }
300  return false;
301  }

386  /**
387  * @brief Attempts to get a lock on the file handle $fp within the specified
388  * timeout, specified by $timeout.
389  *
390  * @param $fp File handle to obtain lock on.
391  * @param $lock Type of lock exclusive or shared (LOCK_EX or LOCK_SH).
392  * @param $timeout Time to wait before giving up on lock in milliseconds.
393  *
394  * @return True if lock obtained, false otherwise
395  *
396  * @todo Find a better location for this, a FileUtils class or something similar.
397  */
398  protected function getLock(&$fp, $lock, $timeout = self::LOCK_WAIT_TIMEOUT)
399  {
400  // Validate lock type
401  if ($lock != LOCK_EX && $lock != LOCK_SH) {
402  return false;
403  }
404  $startTime = microtime(true);
405  do {
406  $canWrite = flock($fp, $lock | LOCK_NB);
407  // If lock not obtained sleep for 0 - 10 milliseconds, to avoid collision and CPU load
408  if (!$canWrite) {
409  $sleep = round(rand(0, 10) * 1000);
410  usleep($sleep);
411  }
412  } while ((!$canWrite) && ((microtime(true) - $startTime) < ($timeout / 1000)));
413  return $canWrite;
414  }
