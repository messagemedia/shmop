96  /**
97  * @brief Create a pack format based on a structure for packing or unpacking.
98  *
99  * \a $structure is expected to be an associative array of name => type pairs.
100  *
101  * \a $id can be any string unique to the \a $structure parameter. It is used to
102  * cache the results, such that calling this function a second time with the
103  * \a $id and \a $named values, will result in this function returning an array
104  * cached from the previous call. This is necessary to prevent a potential (though
105  * very small) performance overhead in the metrics classes.
106  *
107  * @param $id An ID to give to this structure so the returned value
108  * can be stored for fast retrival in subsequent calls.
109  * @param $structure The structure of the data being packed or unpacked.
110  * @param $named Whether or not to name each value, ie for use with unpack.
111  *
112  * @return Format string to be used in pack() and unpack() functions.
113  *
114  * @see http://php.net/manual/en/function.pack.php
115  * @see http://php.net/manual/en/function.unpack.php
116  */
117  public static function getPackFormat($id, $structure, $named = false)
118  {
119  if (isset(self::$packFormats[$id][$named])) {
120  return self::$packFormats[$id][$named];
121  }
122  $format = '';
123  foreach ($structure as $item => $type) {
124  $format .= $type;
125  if ($named) {
126  $format .= $item . '/';
127  }
128  }
129  if ($named) {
130  $format = substr($format, 0, strlen($format) - 1);
131  self::$packFormats[$id][$named] = $format;
132  return $format;
133  }
134  self::$packFormats[$id][$named] = $format;
135  return $format;
136  }
