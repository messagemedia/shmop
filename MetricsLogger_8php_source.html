28 /**
29  * @brief Provides a structure for implementing counter and timer style application
30  * level metrics within PHP, using shared memory for storage.
31  *
32  * This class serves two purposes.
33  *
34  * Firstly is implements functionality required by the SharedMemoryOp class of
35  * which it extends, including an index structure, a method for finding values
36  * in the index, a method for adding values to the index and __get and __set
37  * functions which use shmop_read and shmop_write. The index structure is designed
38  * to allow PMDAs written for PCP to map metrics to metrics defined within the PMDA.
39  * This is done by storing a cluster, item and instance value in the index, which
40  * are the three identifiers PCP uses to index metrics.
41  *
42  * Secondly, it provides a structure for counting and timing metrics. Counter metrics
43  * are simple monotonic counts of something (requests, bytes, etc...), whilst
44  * timing metrics record how long something took (service_time), a histogram of
45  * timings recorded within specific ranges and a count of the number of timings
46  * have been recorded. When a counter metric is specificed, a single metric is created
47  * to capture this metric, when a timing metric is specified, eight metrics are
48  * created.
49  *
50  * Derived classes need to provide a list of metrics to capture with meta data for
51  * each metric including:
52  * - type: counter or timer
53  * - name: name for the metric
54  * - pcp_cluster: a cluster ID which matches the cluster ID given to the metric
55  * in the consuming PMDA
56  * - pcp_item: an item ID which matches the item ID given to the metric
57  * in the consuming PMDA (optional for timer metrics)
58  * - pcp_instance: an instance ID which matches the instance ID given to the metric
59  * in the consuming PMDA, if a metric doesn't belong to an instance
60  * use -1, provided by the constant INSTANCE_DOMAIN_NULL.
61  *
62  * Metrics are provided as an array of associative arrays and are validated upon
63  * construction - in development mode only - according to the following rules:
64  * - Metrics must have a valid type, COUNTER or TIMER
65  * - Metrics must have a unique valid name
66  * - Metrics must have a valid pcp_cluster value between 0 and 65535
67  * - Metrics that aren't TIMERs must have a valid pcp_item value between 0 and 65535
68  * - TIMER metrics without a pcp_item or pcp_start_item value default to pcp_item 0
69  * - Metrics may have a pcp_instance value between -2147483648 and 2147483647,
70  * if no pcp_instance value is provided, -1 is used as a default.
71  * - The combination of a metrics pcp_cluster, pcp_item and pcp_instance values
72  * must be unique within a derived class. Metrics with duplicate values are
73  * ignored.
74  *
75  * Metric values are validated during assignment operations, and will be set to 0 if
76  * invalid values are detected. COUNTER and TIMER metric values must be:
77  * - Positive integer values
78  * - Must not exceed the maximum value for a 32 bit integer value in PHP which
79  * is typically 2147483647 on a 32 bit OS and 4294967295 on a 64 bit OS.
80  *
81  * Example 1 - Defining and incrementing a counter called things.
82  * @code
83  * $metrics = array(
84  * array(
85  * 'type' => MetricsLogger::METRIC_TYPE_COUNTER,
86  * 'name' => 'things',
87  * 'pcp_cluster' => 0,
88  * 'pcp_item' => 0,
89  * 'pcp_instance' => MetricsLogger::INSTANCE_DOMAIN_NULL,
90  * )
91  * );
92  * $object = new DerivedMetricsLogger($metrics);
93  * echo $object->things; // Prints 0
94  *
95  * $object->things = 10; // Set the value of things to 10
96  * echo $object->things; // Prints 10
97  *
98  * $object->things++; // Increment the value of things by 1
99  * echo $object->things; // Prints 11
100  * @endcode
101  *
102  * Example 2 - Defining and recording timings in a timer called time.
103  * @code
104  * $metrics = array(
105  * array(
106  * 'type' => MetricsLogger::METRIC_TYPE_TIMER,
107  * 'name' => 'time',
108  * 'pcp_cluster' => 0,
109  * 'pcp_item' => 0,
110  * 'pcp_instance' => MetricsLogger::INSTANCE_DOMAIN_NULL,
111  * )
112  * );
113  * $object = new DerivedMetricsLogger($metrics);
114  * // On construction, 8 metrics have been created.
115  * echo $object->time.service_time; // Prints 0
116  * echo $object->time.time_taken_0; // Prints 0
117  * echo $object->time.time_taken_1; // Prints 0
118  * echo $object->time.time_taken_2; // Prints 0
119  * echo $object->time.time_taken_3; // Prints 0
120  * echo $object->time.time_taken_4; // Prints 0
121  * echo $object->time.time_taken_5; // Prints 0
122  * echo $object->time.timings_count; // Prints 0
123  *
124  * $object->timing('time', 2000); // Record that a 'time' event took 2 seconds (2000 milliseconds)
125  * echo $object->time.service_time; // Prints 2000, service time is an aggregate count of all timings recorded
126  * echo $object->time.time_taken_0; // Prints 0
127  * echo $object->time.time_taken_1; // Prints 1, 2 seconds falls into this histogram bin
128  * echo $object->time.time_taken_2; // Prints 0
129  * echo $object->time.time_taken_3; // Prints 0
130  * echo $object->time.time_taken_4; // Prints 0
131  * echo $object->time.time_taken_5; // Prints 0
132  * echo $object->time.timings_count; // Prints 1
133  *
134  * $object->timing('time', 15000); // Record that a 'time' event took 15 seconds (15000 milliseconds)
135  * echo $object->time.service_time; // Prints 17000, service time is an aggregate count of all timings recorded
136  * echo $object->time.time_taken_0; // Prints 0
137  * echo $object->time.time_taken_1; // Prints 1
138  * echo $object->time.time_taken_2; // Prints 0
139  * echo $object->time.time_taken_3; // Prints 1, 15 seconds falls into this histogram bin
140  * echo $object->time.time_taken_4; // Prints 0
141  * echo $object->time.time_taken_5; // Prints 0
142  * echo $object->time.timings_count; // Prints 2
143  * @endcode
144  *
145  * Example 3 - Defining duplicate metrics.
146  * @code
147  * $metrics = array(
148  * array(
149  * 'type' => MetricsLogger::METRIC_TYPE_COUNTER,
150  * 'name' => 'first',
151  * 'pcp_cluster' => 0,
152  * 'pcp_item' => 0,
153  * 'pcp_instance' => 0,
154  * ),
155  * array(
156  * 'type' => MetricsLogger::METRIC_TYPE_COUNTER,
157  * 'name' => 'second',
158  * 'pcp_cluster' => 0,
159  * 'pcp_item' => 0,
160  * 'pcp_instance' => 0,
161  * ),
162  * );
163  * $object = new DerivedMetricsLogger($metrics);
164  * echo $object->first; // Prints 0
165  * echo $object->second; // Prints nothing, value is false.
166  * @endcode
167  *
168  * @see api/xml/SoapXmlMetricsLogger.class.php, an implementation of MetricsLogger.
169  * @see http://oss.sgi.com/projects/pcp/
170  */

214  *

216  * and keys and sets the version.
217  *
218  * @param $name Name given to this object, typically an identifier
219  * for the metrics being collected, i.e. soapxml, email2sms, db.
220  * @param $metrics Array of metrics defined for this metrics logger, which
221  * correlate to the metrics defined in the PMDA reading
222  * the data. @see http://linux.die.net/man/3/pmda.
223  * @param $version The version of the data structure being stored in $keyIds.
224  * @param $mode The mode in which this class should operate, read and write or read only.

227  public function __construct(
228  $name,
229  $metrics,
230  $version,

237  $metricsCount = count($this->metrics) * 4; // Allowing for a 4-fold increase in amount of metrics.

245  /**
246  * @brief Creates a key to ID map of all metrics.
247  *
248  * The returned map will contain all metrics that this metrics logged can
249  * keep track of. Map keys will be the name of the metric, map values will be
250  * and associative array containting flags, type, pcp_cluster, pcp_item and
251  * pcp_instance.
252  *
253  * If development mode is enabled, this function will validate all metrics
254  * and warn on duplicate metrics.
255  *
256  * @param $metrics Array of metric configurations.
257  *
258  * @return Associative array of metrics to IDs.
259  */
260  protected function parseMetrics($metrics)
261  {
262  $parsedMetrics = array();
263  if ($this->developmentMode) {
264  $pcpHashes = array(); ///< Used to check for duplicate PCP IDs.
265  }
266  foreach ($metrics as $index => $config) {
267  if ($this->developmentMode) {
268  $this->validateMetricConfig($index, $config, $pcpHashes);
269  }
270  if ($config['type'] == self::METRIC_TYPE_COUNTER) {
271  $keyName = $config['name'];
272  if ($this->developmentMode && isset($parsedMetrics[$keyName])) {
273  $this->logger->warning('Duplicate key name for ' . $keyName);
274  }
275  $parsedMetrics[$keyName] = array(
276  'flags' => 0,
277  'type' => Packing::UINT32,
278  'pcp_cluster' => $config['pcp_cluster'],
279  'pcp_item' => $config['pcp_item'],
280  'pcp_instance' => $config['pcp_instance'],
281  );
282  } elseif ($config['type'] == self::METRIC_TYPE_TIMER) {
283  foreach ($this->timingMetrics as $metric => $type) {
284  $keyName = $config['name'] . '.' . $metric;
285  if ($this->developmentMode && isset($parsedMetrics[$keyName])) {
286  $this->logger->warning('Duplicate key name for ' . $keyName);
287  }
288  $parsedMetrics[$keyName] = array(
289  'flags' => 0,
290  'type' => $type,
291  'pcp_cluster' => $config['pcp_cluster'],
292  'pcp_item' => $config['pcp_item'],
293  'pcp_instance' => $config['pcp_instance'],
294  );
295  $config['pcp_item']++;
296  }
297  }
298  }
299  return $parsedMetrics;
300  }

302  /**
303  * @brief Validation configuration for metrics.
304  *
305  * Ensures metric configurations are valid. All metrics require a pcp_cluster
306  * value, used when creating an ID comprising of the pcp_cluster, pcp_item
307  * and pcp_instance.
308  *
309  * Counter metrics require a valid name.
310  *
311  * Non timer metrics require a pcp_item value, used when creating an ID
312  * comprising of the pcp_cluster, pcp_item and pcp_instance.
313  *
314  * Metrics that don't have a pcp_item value will have this value initialized
315  * to 0, or to the pcp_start_item.
316  *
317  * To minimise the performance impact this class has on code that uses it,
318  * this function is only used in development mode.
319  *
320  * @param $index Index of the item, used for logging warnings.
321  * @param $config Associative array of config values.
322  * @param $pcpHashes Array of hash PCP metrics, used to identify duplicates,
323  * passed by reference.
324  *
325  * @return True if valid, false otherwise.
326  */
327  protected function validateMetricConfig($index, $config, &$pcpHashes)
328  {
329  // Metric must have a valid type
330  if (!isset($config['type']) || !in_array($config['type'], array(
331  self::METRIC_TYPE_COUNTER,
332  self::METRIC_TYPE_TIMER,
333  ))) {
334  $this->logger->warning('Missing or invalid type property, ignoring metric at index ' . $index);
335  return false;
336  }
337 
338  // All metrics must have a name
339  if (!isset($config['name']) || !is_string($config['name'])) {
340  $this->logger->warning('Missing name property, ignoring metric at index ' . $index);
341  return false;
342  }
343 
344  // All metrics must have a pcp_cluster property
345  if (!isset($config['pcp_cluster']) || !is_int($config['pcp_cluster'])) {
346  $this->logger->warning('Missing or invalid pcp_cluster property, ignoring metric at index ' . $index);
347  return false;
348  }
349 
350  if ($config['type'] != self::METRIC_TYPE_TIMER) {
351  // Only timers don't require a pcp_item property
352  if (!isset($config['pcp_item']) || !is_int($config['pcp_item'])) {
353  $this->logger->warning('Missing or invalid pcp_item property, ignoring metric at index ' . $index);
354  return false;
355  }
356  }
357 
358  if ($config['pcp_cluster'] < 0 || $config['pcp_cluster'] > 65535) { // Max UINT16
359  $this->logger->warning('Out of range pcp_cluster property, ignoring metric at index ' . $index);
360  return false;
361  }
362  if ($config['pcp_item'] < 0 || $config['pcp_item'] > 65535) { // Max UINT16
363  $this->logger->warning('Out of range pcp_item property, ignoring metric at index ' . $index);
364  return false;
365  }
366  if ($config['pcp_instance'] < -2147483648 || $config['pcp_instance'] > 2147483647) { // Max INT32
367  $this->logger->warning('Out of range pcp_instance property, ignoring metric at index ' . $index);
368  return false;
369  }
370 
371  $pcpHash = md5($config['pcp_cluster'] . $config['pcp_item'] . $config['pcp_instance']);
372  if (in_array($pcpHash, $pcpHashes)) {
373  $this->logger->warning('Duplicate PCP ID, ignoring metric at index ' . $index);
374  return false;
375  } else {
376  $pcpHashes[] = $pcpHash;
377  }
378  return true;
379  }

381  /**
382  * @brief Wraps findValue() function with shared locking.
383  *
384  * @param $name Name of value to find in the index shared memory segment.
385  *
386  * @return True if value found, false otherwise.
387  */
388  protected function findValueWithLock($name)
389  {
390  if (!isset($this->metrics[$name])) {
391  $this->logger->warning('Attempted to find undefined metric ' . $name);
392  return false;
393  }
394 
395  $fp = fopen($this->ipcKeyFile, 'r');
396  if ($fp !== false) {
397  try {
398  if ($this->getLock($fp, LOCK_SH)) {
399  $valueFound = $this->findValue($name);
400  flock($fp, LOCK_UN);
401  fclose($fp);
402  return $valueFound;
403  } else {
404  $this->logger->error('Failed to obtain a lock on ' . $this->ipcKeyFile);
405  }
406  } catch (Exception $e) {
407  // Let any exceptions fall through, unlock and close the file
408  $this->logger->error($e->getMessage());
409  }
410  flock($fp, LOCK_UN);
411  fclose($fp);
412  } else {
413  $this->logger->error('Failed to open ' . $this->ipcKeyFile);
414  }
415  return false;
416  }

418  /**
419  * @brief Attempts to find a value in the index shared memory segment to
420  * get its offset in the data shared memory segment.
421  *
422  * This function should only be used if a lock has been obtained on the
423  * IPC key file. Using findValueWith lock provides this functionality.
424  *
425  * @param $name Name of value to find in the index shared memory segment.
426  *
427  * @return True if value found, false otherwise.
428  */
429  protected function findValue($name)
430  {
431  $cluster = $this->metrics[$name]['pcp_cluster'];
432  $item = $this->metrics[$name]['pcp_item'];
433  $instance = $this->metrics[$name]['pcp_instance'];
434  $head = $this->getHead();
435 
436  $startIndexOffset = Packing::getPackLength($this->getHeadStructure()); // Indexes are after the head structure.
437  $endIndexOffset = $head['nextIndexOffset']; // nextIndexOffset gives the tail, just past the final index item.
438  $indexStructureLength = Packing::getPackLength($this->indexStructure);
439 
440  for ($indexOffset = $startIndexOffset; $indexOffset < $endIndexOffset; $indexOffset += $indexStructureLength) {
441  $data = unpack(
442  Packing::getPackFormat('index', $this->indexStructure, true),
443  shmop_read($this->shmIndexId, $indexOffset, $indexStructureLength)
444  );
445  if (($data['cluster'] == $cluster) && ($data['item'] == $item) && ($data['instance'] == $instance)) {
446  $this->indexData[$name]['offset'] = $data['offset'];
447  return true;
448  }
449  }
450  return false;
451  }

453  /**
454  * @brief Adds a new value to the shared memory segment.
455  *
456  * Adds a value to the shared memory segment index and this classes internal
457  * index.
458  *
459  * @param $name Name of the value to add.
460  *
461  * @return True if added, false otherwise.
462  */
463  protected function addNewValue($name)
464  {
465  if (!isset($this->metrics[$name])) {
466  $this->logger->warning('Attempted to add undefined metric ' . $name);
467  return false;
468  }
469 
470  $fp = fopen($this->ipcKeyFile, 'r');
471  if ($fp !== false) {
472  try {
473  if ($this->getLock($fp, LOCK_EX)) {
474  if ($this->findValue($name)) {
475  // The value may have been added between the last check
476  // and locking of the file.
477  return true;
478  }
479 
480  $head = $this->getHead();
481  $indexStructureLength = Packing::getPackLength($this->indexStructure);
482 
483  if (shmop_size($this->shmIndexId) < $head['nextIndexOffset'] + $indexStructureLength) {
484  $this->logger->error('Index shared memory segment full, cannot add ' . $name);
485  return false;
486  }
487 
488  $newValueLength = Packing::getTypeLength($this->metrics[$name]['type']);
489  if (shmop_size($this->shmDataId) < $head['nextDataOffset'] + $newValueLength) {
490  $this->logger->error('Data shared memory segment full, cannot add ' . $name);
491  return false;
492  }
493 
494  /// Zero the destination data.
495  shmop_write($this->shmDataId, pack("x{$newValueLength}"), $head['nextDataOffset']);
496 
497  $cluster = $this->metrics[$name]['pcp_cluster'];
498  $item = $this->metrics[$name]['pcp_item'];
499  $instance = $this->metrics[$name]['pcp_instance'];
500  $type = $this->metrics[$name]['type'];
501 
502  $flags = 0; // Currently unused
503 
504  // Write the new item into the index.
505  $data = pack(
506  Packing::getPackFormat('index', $this->indexStructure),
507  $flags, ///< Order of parameters must match order in $this->indexStrucutre
508  ord($type), ///< @todo Find a way to pass these in the same order as $this->indexStrucutre
509  $newValueLength,
510  $head['nextDataOffset'],
511  $cluster,
512  $item,
513  $instance
514  );
515  if (strlen($data) != $indexStructureLength) {
516  $this->logger->error(
517  'Incorrect index length ' . strlen($data) . ', length should be ' . $indexStructureLength
518  );
519  return false;
520  }
521  $bytesWritten = shmop_write($this->shmIndexId, $data, $head['nextIndexOffset']);
522  if ($bytesWritten != $indexStructureLength) {
523  $this->logger->error(
524  'Incorrect index bytes written ' . $bytesWritten . ', bytes written should be ' .
525  $indexStructureLength
526  );
527  return false;
528  }
529 
530  // Store data about this value locally.
531  $this->indexData[$name]['offset'] = $head['nextDataOffset'];
532  $this->indexData[$name]['length'] = $newValueLength;
533  $this->indexData[$name]['type'] = $type;
534 
535  // Write the new offsets to the index segment.
536  $head['nextIndexOffset'] += $bytesWritten;
537  $head['nextDataOffset'] += $newValueLength;
538  $packedOffsetData = pack(
539  Packing::getPackFormat('offset', $this->offsetsStructure),
540  $head['nextIndexOffset'],
541  $head['nextDataOffset']
542  );
543  $packedOffsetDataBytesWritten = shmop_write(
544  $this->shmIndexId,
545  $packedOffsetData,
546  Packing::getPackLength($this->versionStructure)
547  );
548  if ($packedOffsetDataBytesWritten == Packing::getPackLength($this->offsetsStructure)) {
549  flock($fp, LOCK_UN);
550  fclose($fp);
551  return true;
552  }
553  } else {
554  $this->logger->error('Failed to obtain a lock on ' . $this->ipcKeyFile);
555  }
556  } catch (Exception $e) {
557  // Let any exceptions fall through, unlock and close the file
558  $this->logger->error($e->getMessage());
559  }
560  flock($fp, LOCK_UN);
561  fclose($fp);
562  } else {
563  $this->logger->error('Failed to open ' . $this->ipcKeyFile);
564  }
565  return false;
566  }

568  /**
569  * @brief Validates a value against its type and modifies
570  * if necessary.
571  *
572  * @param $name Name of the value to validate.
573  * @param $value The value to validate.
574  */
575  protected function validateValue(&$value, $name)
576  {
577  // All metrics should be numeric.
578  if (is_numeric($value)) {
579  // Ensure integers don't exceed their limit and convert to doubles.
580  $maxUint32Value = min(PHP_INT_MAX, 4294967295); // The former applies on 32-bit systems.
581  if ($this->metrics[$name]['type'] == Packing::UINT32 && ($value >= $maxUint32Value || !is_int($value))) {

598  *
599  * Reads a value from the shared memory segment initialized by this class.
600  * When getting a value whose name doesn't exist in the keyIds array, false
601  * is returned. False is also returned if the shared memory segment cannot
602  * be accessed or read from.
603  *
604  * @param $name Name of the value to return.
605  *
606  * @return Value of that property if it's name exists in the classes $metrics
607  * array, false if it doens't exist or the shared memory segment
608  * cannot be accessed or read from.
609  */
610  public function __get($name)
611  {
612  if ($this->hasError) {
613  return false;
614  }
615  if ($this->shmIndexId === false || $this->shmDataId === false) {
616  if (!$this->initializeSharedMemorySegment()) {
617  $this->hasError = true;
618  }
619  }
620  if ($this->hasError) {
621  return false;
622  }
623  if ((!array_key_exists($name, $this->indexData)) && (!$this->findValueWithLock($name))) {
624  // If the value has explicitly defined as a metric, return 0
625  if (array_key_exists($name, $this->metrics)) {
626  return 0;
627  }
628  return false;
629  }
630  $data = unpack(
631  "{$this->metrics[$name]['type']}value",
632  shmop_read(
633  $this->shmDataId,
634  $this->indexData[$name]['offset'],
635  Packing::getTypeLength($this->metrics[$name]['type'])
636  )
637  );
638  return $data['value'];
639  }

716  *
717  * This function will update a number of metrics including:
718  * - service_time, an aggregate of the total time recorded for this key.
719  * - responses_times_n, a histogram bin (0 - 5) counting response times
720  * @see: $timingMetrics.
721  * - count, the number of times this key has been recorded.
722  *
723  * @param $key Key to update metrics for
724  * @param $timeTaken Duration for the event in milliseconds
725  */
726  public function timing($key, $timeTaken)
727  {
728  if ($this->readOnly) {
729  $this->logger->error('Attempted to write in read only mode');
730  }
731  if ($this->hasError) {
732  return;
733  }
734  $this->{$key . '.service_time'} += $timeTaken;
735  switch (true) {
736  case ($timeTaken < 1000): // Less than 1 second
737  $this->{$key . '.time_taken_0'}++;
738  break;
739  case ($timeTaken >= 1000 && $timeTaken < 5000): // Between 1 and 5 seconds
740  $this->{$key . '.time_taken_1'}++;
741  break;
742  case ($timeTaken >= 5000 && $timeTaken < 10000): // Between 5 and 10 seconds
743  $this->{$key . '.time_taken_2'}++;
744  break;
745  case ($timeTaken >= 10000 && $timeTaken < 20000): // Between 10 and 20 seconds
746  $this->{$key . '.time_taken_3'}++;
747  break;
748  case ($timeTaken >= 20000 && $timeTaken < 40000): // Between 20 and 40 seconds
749  $this->{$key . '.time_taken_4'}++;
750  break;
751  case ($timeTaken >= 40000): // Greater than 40 seconds
752  $this->{$key . '.time_taken_5'}++;
753  break;
754  }

756  }
